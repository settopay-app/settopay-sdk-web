// 1. SDK 초기화 SettoSDK.initialize({ environment: 'dev', // idpToken: 'firebase-id-token', // 선택: 있으면 자동로그인 debug: true, }); // 2. 결제 요청 const result = await SettoSDK.openPayment({ merchantId: '01KDHYY1EAQ98FHG1TQ07KHKHW', amount: '10', orderId: 'order-123', // 선택 }); // 3. 결과 처리 if (result.status === 'success') { console.log('Payment ID:', result.paymentId); console.log('Tx Hash:', result.txHash); } else if (result.status === 'failed') { console.error('Error:', result.error); } else { console.log('Cancelled'); }
